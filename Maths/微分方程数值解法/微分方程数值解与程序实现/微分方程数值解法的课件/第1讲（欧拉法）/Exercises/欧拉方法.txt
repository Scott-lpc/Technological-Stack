#include "stdafx.h"
#include<math.h>
#include<stdio.h>
#include<stdlib.h>
void main( )
{
  double a,b,dx;
  int i,N;
  double f(double x, double y);
  double exact(double x);
  double *x;
  double *y;


  a=0.0;
  b=0.1;
  N=5;
  dx=(b-a)/N;
  x=(double *)malloc(sizeof(double)*(N+1));
  y=(double *)malloc(sizeof(double)*(N+1));
  y[0]=1.0;
  for(i=0;i<=N;i++)
  {
    x[i]=a+i*dx;
    if(i!=N)
		y[i+1]=y[i]+dx*f(x[i],y[i]);
    printf("x=%.2f,   numerical=%.6f,  exact=%.6f,   error=%.6f\n",x[i],y[i],exact(x[i]),fabs(y[i]-exact(x[i])));
  }
  free(x);
  free(y);
}

double f(double x, double y)
{
  double z;
  z=-0.9*y/(1+2*x);
  return z;
}

double exact(double x)
{
  double z;
  z=pow((1.0+2*x),-0.45);
  return z;
}
